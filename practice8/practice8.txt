#include <stdlib.h>
#include <math.h>
#define endmark -1
double stream xy;
double stream xz;
double stream yx;
double stream yz;
double stream zx;
double stream zy;

boolean isConverage(double a,double b){
    return  fabs(a-b)<0.0001;
}

void  sloveX(){
    double newx,x,y,z;
    recv(yx,y);
    recv(zx,z);
    while(true){
        newx =(8+z-y)/2;
        if(isConverage(newx,x)){
            x=newx;
            send(xy,x);
            send(xz,x);
            recv(yx,y);
            recv(zx,z);
        }
        else{
            break;
        }
    }
    cout << "x:="<< x << endl;
}

void  sloveY(){
    double x,newy,y,z;
    recv(zy,z);
    recv(xy,x);
     while(true){
        newy =-1*(-11-2*z+3*x);
        if(isConverage(newy,y)){
            y=newy;
            send(yz,y);
            send(yx,y);
            recv(zy,z);
            recv(xy,x);
        }
        else{
            break;
        }
    }
    cout << "y:="<< y << endl;
}

void  sloveZ(){
    double x,y,newz,z;
    recv(yz,y);
    recv(xz,x);
    while(true){
        newz = -1*(-3-2*x+-y)/2;
        if(isConverage(newz,z)){
            z=newz;
            send(xz,z);
            send(xy,z);
            recv(yz,y);
            recv(xz,x);
        }
        else{
            break;
        }
    }
   
    cout << "z:="<< z << endl;
}
